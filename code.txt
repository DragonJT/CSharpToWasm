
static class Program
{
    [import]
    static void Print(int i)
    {
        console.log("int "+i);
    }

    [import]
    static void Print(float f)
    {
        console.log("float "+f);
    }

    [export]
    static void Main()
    {
        int[] z = [32, 45, 56, 73];
        Print(z[0]);
        Print(z[1]);
        Print(z[2]);
        Print(z[3]);
    }
}
